    <script>
        let cachedSourceFileName = "bg_removed.png";

        // Drag-and-Drop Surface UI Styling Tweaks
        const dropZone = document.getElementById('drop-zone');

        ['dragenter', 'dragover'].forEach(eventName => {
            dropZone.addEventListener(eventName, (e) => {
                e.preventDefault();
                dropZone.classList.add('dragover');
            }, false);
        });

        ['dragleave', 'drop'].forEach(eventName => {
            dropZone.addEventListener(eventName, (e) => {
                e.preventDefault();
                dropZone.classList.remove('dragover');
            }, false);
        });

        dropZone.addEventListener('drop', (e) => {
            const dt = e.dataTransfer;
            const files = dt.files;
            if (files.length > 0 && !document.getElementById('upload-input').disabled) {
                handleFileUpload(files[0]);
            }
        });

        // Pipeline Handler Core Ingestion Engine Matrix
        async function handleFileUpload(file) {
            if (!file || !file.type.startsWith('image/')) {
                alert('Please provide a valid image asset configuration profile.');
                return;
            }

            cachedSourceFileName = file.name.substring(0, file.name.lastIndexOf('.')) + '_no_bg.png';

            // Swap visual card layers smoothly 
            document.getElementById('section-upload').classList.add('hidden-section');
            document.getElementById('section-loading').classList.remove('hidden-section');

            // Render original source file layer structure preview safely
            const reader = new FileReader();
            reader.onload = (e) => {
                document.getElementById('output-original').src = e.target.result;
            };
            reader.readAsDataURL(file);

            // Execute Neural ML Core Instance Configuration Architecture pipeline parameters
            try {
                const config = {
                    debug: false,
                    device: 'gpu', // WebGPU fallback acceleration framework logic channel
                    output: {
                        format: 'image/png',
                        quality: 1.0
                    },
                    progress: (key, current, total) => {
                        let percent = 0;
                        if (total) {
                            percent = Math.round((current / total) * 100);
                        }
                        
                        document.getElementById('processing-progress-bar').style.width = percent + '%';
                        
                        if (key.includes('fetch')) {
                            document.getElementById('progress-status-text').innerText = `Downloading AI Models: ${percent}%`;
                        } else {
                            document.getElementById('progress-status-text').innerText = `Isolating Foreground Subject...`;
                        }
                    }
                };

                // Trigger WASM engine function call
                const outputBlob = await window.imglyRemoveBackground(file, config);
                
                // Mount blob outputs to UI structures 
                const transparentUrl = URL.createObjectURL(outputBlob);
                document.getElementById('output-processed').src = transparentUrl;
                
                // Map actions safely to browser anchor attributes
                const downloadBtn = document.getElementById('btn-download');
                downloadBtn.href = transparentUrl;
                downloadBtn.download = cachedSourceFileName;

                // View results dashboard 
                document.getElementById('section-loading').classList.add('hidden-section');
                document.getElementById('section-results').classList.remove('hidden-section');

            } catch (error) {
                console.error("AI Core Processing Exception Trace Block Details: ", error);
                alert("An error occurred during localized subject extraction segmentation. Please try re-uploading a smaller file format alternative.");
                resetWorkspace();
            }
        }

        // Workspace reset trigger parameters
        function resetWorkspace() {
            document.getElementById('upload-input').value = "";
            document.getElementById('processing-progress-bar').style.width = '0%';
            document.getElementById('output-original').src = "";
            document.getElementById('output-processed').src = "";
            
            document.getElementById('section-results').classList.add('hidden-section');
            document.getElementById('section-loading').classList.add('hidden-section');
            document.getElementById('section-upload').classList.remove('hidden-section');
        }
    </script>
